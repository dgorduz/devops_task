# Create a new local user account for running Jenkins
$userName = 'jenkins'
$userPassword = 'P@ssw0rd' | ConvertTo-SecureString -AsPlainText -Force
$credential = New-Object System.Management.Automation.PSCredential ($userName, $userPassword)
New-LocalUser -Name $userName -Password $userPassword

# Download the latest Jenkins WAR file
$jenkinsUrl = 'http://updates.jenkins-ci.org/latest/jenkins.war'
$jenkinsPath = 'C:\Program Files (x86)\Jenkins\jenkins.war'
Invoke-WebRequest -Uri $jenkinsUrl -OutFile $jenkinsPath

# Create a new Jenkins service that runs under the jenkins user account
$serviceName = 'Jenkins'
$binaryPath = "C:\Windows\System32\java.exe -jar $jenkinsPath --httpPort=8080"
$serviceParams = @{
    Name = $serviceName
    BinaryPathName = $binaryPath
    Credential = $credential
    Description = 'Jenkins CI server'
}
New-Service @serviceParams
Start-Service $serviceName
